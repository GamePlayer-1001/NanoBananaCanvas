// router/index.tsx
import { createBrowserRouter } from 'react-router-dom'

export const router = createBrowserRouter([
  {
    path: '/',
    element: <RootGuard />,  // 根守卫：根据登录状态分流
    children: [
      // Landing page routes (未登录)
      {
        element: <LandingLayout />,
        children: [
          { index: true, element: <LandingPage /> },
          { path: 'pricing', element: <PricingPage /> },
        ]
      },
    ]
  },
  // App routes (已登录)
  {
    path: '/',
    element: <MainLayout />,
    children: [
      { path: 'workspace', element: <WorkspacePage /> },
      { path: 'explore', element: <ExplorePage /> },
      { path: 'explore/:id', element: <WorkflowDetailPage /> },
      { path: 'profile', element: <ProfilePage /> },
    ]
  },
  {
    path: '/workspace/:id',
    element: <ProtectedRoute><EditorPage /></ProtectedRoute>
  }
])

// RootGuard.tsx - 根路由守卫
function RootGuard() {
  const { isSignedIn, isLoaded } = useAuth()

  if (!isLoaded) return <LoadingSpinner />

  // 已登录 → 重定向到 /workspace
  // 未登录 → 显示 Landing Page
  return isSignedIn
    ? <Navigate to="/workspace" replace />
    : <Outlet />
}

  if (!isSignedIn) return <Navigate to="/" />

  return children

// FirstVisitModal.tsx
function FirstVisitModal() {
  const [show, setShow] = useState(false)
  const { isSignedIn } = useUser()

  useEffect(() => {
    const hasVisited = localStorage.getItem('hasVisited')
    if (!hasVisited && !isSignedIn) {
      setShow(true)
      localStorage.setItem('hasVisited', 'true')
    }
  }, [isSignedIn])

  if (!show) return null

  return (
    <Modal onClose={() => setShow(false)}>
      <SignIn />
    </Modal>
  )
}

### 6.2 后端中间件

```typescript
// middleware/auth.ts
import { verifyToken } from '@clerk/backend'

export async function authMiddleware(c, next) {
  const token = c.req.header('Authorization')?.replace('Bearer ', '')

  if (!token) {
    return c.json({ error: 'Unauthorized' }, 401)
  }

  try {
    const payload = await verifyToken(token, {
      secretKey: c.env.CLERK_SECRET_KEY
    })
    c.set('userId', payload.sub)
    return next()
  } catch (error) {
    return c.json({ error: 'Invalid token' }, 401)
  }
}

// 可选认证（访客也可访问）
export async function optionalAuthMiddleware(c, next) {
  const token = c.req.header('Authorization')?.replace('Bearer ', '')

  if (token) {
    try {
      const payload = await verifyToken(token, {
        secretKey: c.env.CLERK_SECRET_KEY
      })
      c.set('userId', payload.sub)
    } catch {
      // 忽略无效 token
    }
  }

  return next()
}
```
