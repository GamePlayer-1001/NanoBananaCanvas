## 一、设计目标

为用户上传（图片作为 img2video 输入）、AI 生成结果（图片/视频/音频）、工作流缩略图等文件提供统一的上传、存储、分发、清理方案。

## 二、R2 存储结构

### 2.1 Bucket 目录设计

```
nano-banana-canvas-files/
├── uploads/                          # 用户上传的原始文件
│   └── {user_id}/
│       └── {file_id}.{ext}          # UUID 命名，避免冲突
│
├── outputs/                          # AI 生成结果
│   └── {user_id}/
│       └── {task_id}.{ext}          # 与 async_tasks.id 对应
│
├── thumbnails/                       # 工作流缩略图
│   └── {workflow_id}.webp            # 统一 webp 格式，256x256
│
└── avatars/                          # 用户头像（Clerk 同步备份）
    └── {user_id}.webp
```

### 2.2 命名规范

| 文件类型 | Key 格式 | 示例 |
|---------|---------|------|
| 用户上传 | `uploads/{userId}/{fileId}.{ext}` | `uploads/user_abc/f1a2b3.png` |
| AI 输出 | `outputs/{userId}/{taskId}.{ext}` | `outputs/user_abc/task_xyz.mp4` |
| 缩略图 | `thumbnails/{workflowId}.webp` | `thumbnails/wf_123.webp` |
| 头像 | `avatars/{userId}.webp` | `avatars/user_abc.webp` |

## 三、上传流程

### 3.1 预签名 URL 上传（推荐）

```
前端                        Next.js API                     R2
 │                              │                             │
 │  1. POST /api/files/upload   │                             │
 │     { filename, type, size } │                             │
 │  ─────────────────────────►  │                             │
 │                              │  2. 校验文件类型/大小         │
 │                              │  3. 生成 R2 预签名 URL       │
 │                              │  ─────────────────────────►  │
 │                              │  ◄─────────────────────────  │
 │  4. { uploadUrl, fileKey }   │                             │
 │  ◄─────────────────────────  │                             │
 │                              │                             │
 │  5. PUT uploadUrl (binary)   │                             │
 │  ──────────────────────────────────────────────────────►   │
 │                              │                             │
 │  6. 上传完成，前端持有 fileKey │                            │
 │     传给 AI 节点作为输入      │                             │
```

### 3.2 API 实现

```typescript
// app/api/files/upload/route.ts

import { auth } from '@clerk/nextjs/server'
import { NextRequest, NextResponse } from 'next/server'

/* ============================================ */
/*  文件上传预签名 URL 生成                       */
/*  [INPUT]: Clerk auth, request body            */
/*  [OUTPUT]: R2 presigned URL + file key        */
/* ============================================ */

const ALLOWED_TYPES: Record<string, string[]> = {
  image: ['image/png', 'image/jpeg', 'image/webp', 'image/gif'],
  video: ['video/mp4', 'video/webm'],
  audio: ['audio/mp3', 'audio/wav', 'audio/mpeg'],
}

const MAX_FILE_SIZE: Record<string, number> = {
  image: 10 * 1024 * 1024,     // 10MB
  video: 100 * 1024 * 1024,    // 100MB
  audio: 20 * 1024 * 1024,     // 20MB
}

export async function POST(request: NextRequest) {
  const { userId } = await auth()
  if (!userId) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  const { filename, contentType, size, category } = await request.json()

  // 校验文件类型
  const allowedTypes = ALLOWED_TYPES[category]
  if (!allowedTypes?.includes(contentType)) {
    return NextResponse.json(
      { error: `File type ${contentType} not allowed for ${category}` },
      { status: 400 }
    )
  }

  // 校验文件大小
  const maxSize = MAX_FILE_SIZE[category]
  if (size > maxSize) {
    return NextResponse.json(
      { error: `File size exceeds ${maxSize / 1024 / 1024}MB limit` },
      { status: 400 }
    )
  }

  // 生成文件 Key
  const ext = filename.split('.').pop() ?? 'bin'
  const fileId = crypto.randomUUID()
  const fileKey = `uploads/${userId}/${fileId}.${ext}`

  // 生成预签名 URL（有效期 10 分钟）
  const uploadUrl = await generatePresignedUrl(fileKey, contentType)

  return NextResponse.json({
    uploadUrl,
    fileKey,
    expiresIn: 600,
  })
}

async function generatePresignedUrl(
  key: string,
  contentType: string
): Promise<string> {
  // Cloudflare R2 S3 兼容 API 生成预签名 URL
  // 使用 @aws-sdk/s3-request-presigner 或 R2 Worker binding
  const { S3Client, PutObjectCommand } = await import('@aws-sdk/client-s3')
  const { getSignedUrl } = await import('@aws-sdk/s3-request-presigner')

  const s3 = new S3Client({
    region: 'auto',
    endpoint: process.env.R2_ENDPOINT,
    credentials: {
      accessKeyId: process.env.R2_ACCESS_KEY_ID!,
      secretAccessKey: process.env.R2_SECRET_ACCESS_KEY!,
    },
  })

  const command = new PutObjectCommand({
    Bucket: process.env.R2_BUCKET_NAME,
    Key: key,
    ContentType: contentType,
  })

  return getSignedUrl(s3, command, { expiresIn: 600 })
}
```

### 3.3 前端上传 Hook

```typescript
// hooks/use-file-upload.ts

export function useFileUpload() {
  const [uploading, setUploading] = useState(false)
  const [progress, setProgress] = useState(0)

  async function upload(file: File, category: 'image' | 'video' | 'audio') {
    setUploading(true)
    setProgress(0)

    // 1. 获取预签名 URL
    const { uploadUrl, fileKey } = await fetch('/api/files/upload', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        filename: file.name,
        contentType: file.type,
        size: file.size,
        category,
      }),
    }).then(r => r.json())

    // 2. 直接上传到 R2
    await fetch(uploadUrl, {
      method: 'PUT',
      headers: { 'Content-Type': file.type },
      body: file,
    })

    setUploading(false)
    setProgress(100)

    return fileKey
  }

  return { upload, uploading, progress }
}
```

## 四、文件访问

### 4.1 公开文件（CDN 分发）

| 文件类型 | 访问方式 | 缓存策略 |
|---------|---------|---------|
| 缩略图 | R2 Public Bucket + 自定义域名 | `Cache-Control: public, max-age=86400` |
| 公开作品的输出 | R2 Public URL | `Cache-Control: public, max-age=3600` |

### 4.2 私有文件（签名访问）

| 文件类型 | 访问方式 | 有效期 |
|---------|---------|--------|
| 用户上传 | 预签名 GET URL | 1 小时 |
| 未公开的 AI 输出 | 预签名 GET URL | 1 小时 |

```typescript
// app/api/files/[key]/route.ts

export async function GET(
  request: NextRequest,
  { params }: { params: { key: string } }
) {
  const { userId } = await auth()
  if (!userId) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const fileKey = decodeURIComponent(params.key)

  // 权限检查：用户只能访问自己的文件
  if (!fileKey.includes(userId)) {
    return NextResponse.json({ error: 'Forbidden' }, { status: 403 })
  }

  const signedUrl = await generatePresignedGetUrl(fileKey)
  return NextResponse.json({ url: signedUrl })
}
```

## 五、存储配额与清理

### 5.1 按套餐存储配额

| 套餐 | 上传配额 | 输出保留 | 清理策略 |
|------|---------|---------|---------|
| Free | 1 GB | 7 天后删除 | 自动 |
| Standard | 10 GB | 30 天后删除 | 自动 |
| Pro | 50 GB | 90 天后删除 | 自动 |
| Ultimate | 200 GB | 永久保留 | 手动 |

### 5.2 清理任务（Cron Trigger）

```typescript
// apps/worker/src/cron/cleanup.ts

export async function cleanupExpiredFiles(env: Env) {
  const now = Date.now()

  // 查询过期任务
  const expired = await env.DB.prepare(`
    SELECT at.id, at.user_id, at.output_data, s.plan
    FROM async_tasks at
    JOIN subscriptions s ON s.user_id = at.user_id
    WHERE at.status = 'completed'
    AND at.completed_at < ?
    AND s.plan IN ('free', 'standard', 'pro')
  `).bind(now - getRetentionMs('free')).all()

  for (const task of expired.results) {
    const output = JSON.parse(task.output_data)
    if (output.r2_key) {
      await env.BUCKET.delete(output.r2_key)
    }
  }
}

function getRetentionMs(plan: string): number {
  const days: Record<string, number> = {
    free: 7,
    standard: 30,
    pro: 90,
  }
  return (days[plan] ?? 90) * 24 * 60 * 60 * 1000
}
```

## 六、缩略图生成

### 6.1 工作流缩略图

工作流发布到广场时，自动截取画布快照作为缩略图。

```typescript
// 前端生成画布截图
import { toPng } from 'html-to-image'

async function generateThumbnail(canvasElement: HTMLElement): Promise<Blob> {
  const dataUrl = await toPng(canvasElement, {
    width: 512,
    height: 512,
    quality: 0.8,
  })

  // 转为 Blob
  const res = await fetch(dataUrl)
  return res.blob()
}

// 上传缩略图
async function uploadThumbnail(workflowId: string, blob: Blob) {
  const { uploadUrl } = await fetch('/api/files/upload', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      filename: `${workflowId}.png`,
      contentType: 'image/png',
      size: blob.size,
      category: 'image',
      path: `thumbnails/${workflowId}.webp`,
    }),
  }).then(r => r.json())

  await fetch(uploadUrl, {
    method: 'PUT',
    headers: { 'Content-Type': 'image/png' },
    body: blob,
  })
}
```

## 七、安全约束

| 约束 | 实现 |
|------|------|
| 文件类型白名单 | 服务端校验 Content-Type + 文件头 Magic Bytes |
| 文件大小限制 | 预签名 URL 附带 Content-Length 条件 |
| 路径穿越防护 | fileKey 由服务端生成，不接受用户自定义路径 |
| 访问控制 | 预签名 URL 含用户 ID，跨用户访问被拒绝 |
| 恶意文件扫描 | P2 阶段接入 Cloudflare 内容扫描（如需要） |
